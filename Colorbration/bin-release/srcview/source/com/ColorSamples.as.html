package com
{
    public class ColorSamples
    {
        [Bindable]public var nameList:Array;
        [Bindable]public var objList:Array;
        protected var count:int = 0;
        
        public function ColorSamples()
        {
            nameList = new Array();
            objList = new Array();
            onInit();
        }
        
        protected function onInit():void
        {
            onCreate();
        }
        
        public function onCreate(name:String=""):void
        {
            if (!name.length) { 
                name = SpectralData.NAME + count.toString();
                count ++;
            }
            var sample:SpectralData = new SpectralData();
            sample.name = name;
            objList.push(sample);
            nameList.push(sample.name);    
        }
    }
}
