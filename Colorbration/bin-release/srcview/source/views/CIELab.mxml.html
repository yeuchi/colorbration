<?xml version="1.0" encoding="utf-8"?>
<mx:Canvas xmlns:mx="http://www.adobe.com/2006/mxml" 
            width="100%" height="100%" 
            creationComplete="onCreationComplete()">
    
    <mx:Script>
        <![CDATA[
            import com.XYZ2Lab;
            import com.Spectral2XYZ;
            [Bindable] public var wid:int = 80;
            public var oXYZ2Lab:XYZ2Lab;
            
            protected function onCreationComplete():void
            {
                oXYZ2Lab = new XYZ2Lab();
            }
            
            public function set XYZ (sampleXYZ:Spectral2XYZ):void
            {
                oXYZ2Lab.cieXYZ = sampleXYZ;
                if ( !oXYZ2Lab.isEmpty() ) {
                    oXYZ2Lab.forward();
                    txtL.text = oXYZ2Lab.L.toString();
                    txta.text = oXYZ2Lab.a.toString();
                    txtb.text = oXYZ2Lab.b.toString();
                }
            }

        ]]>
    </mx:Script>
    
    <mx:VBox width="100%" height="100%" horizontalAlign="center" verticalAlign="middle">
        <mx:HBox>
            <mx:VBox>
                <mx:Label text="L*"/>
                <mx:TextInput id="txtL" width="{wid}"/>
            </mx:VBox>
            <mx:VBox>
                <mx:Label text="a*"/>
                <mx:TextInput id="txta" width="{wid}"/>
            </mx:VBox>
            <mx:VBox>
                <mx:Label text="b*"/>
                <mx:TextInput id="txtb" width="{wid}"/>
            </mx:VBox>
        </mx:HBox>
    </mx:VBox>
